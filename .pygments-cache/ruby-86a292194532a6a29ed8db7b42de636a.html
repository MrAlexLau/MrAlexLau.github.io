# these loops all execute 5 times

# while loop 
i = 1
while i <= 5
  puts 'this is a while loop'
  i += 1
end 

# do while loop (begin/end/while in ruby)
i = 1
begin
  puts 'this is a do while loop'
  i += 1
end while i <= 5 

# for loop
for i in 1..5
  puts 'this is a for loop'
  i += 1
end

# looping with a times block
5.times do 
  puts 'this is a times block'
end
